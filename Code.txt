import React, { useState, useMemo, useEffect } from 'react';
import { 
  Home, 
  PlusCircle, 
  List, 
  CreditCard, 
  PieChart as PieChartIcon, 
  ArrowUpCircle, 
  ArrowDownCircle, 
  Calendar as CalendarIcon, 
  Tag, 
  FileText,
  Trash2,
  Cloud,
  CloudOff,
  Loader2,
  ChevronDown,
  ChevronUp,
  X,
  Pencil
} from 'lucide-react';

// --- Firebase Imports ---
import { initializeApp } from 'firebase/app';
import { getAuth, signInWithCustomToken, signInAnonymously, onAuthStateChanged } from 'firebase/auth';
import { getFirestore, collection, doc, setDoc, deleteDoc, onSnapshot, updateDoc } from 'firebase/firestore';

// --- Firebase Initialization ---
const firebaseConfig = typeof __firebase_config !== 'undefined' ? JSON.parse(__firebase_config) : null;
const appId = typeof __app_id !== 'undefined' ? __app_id : 'default-app-id';
const initToken = typeof __initial_auth_token !== 'undefined' ? __initial_auth_token : null;

let app, auth, db;
if (firebaseConfig) {
  app = initializeApp(firebaseConfig);
  auth = getAuth(app);
  db = getFirestore(app);
}

// --- Constants ---
const DEFAULT_CATEGORIES = {
  expense: ['餐飲', '交通', '購物', '娛樂', '居家', '其他'],
  income: ['薪資', '投資', '獎金', '其他'],
  installment: ['購物', '家電', '進修', '保險']
};

const CATEGORY_COLORS = {
  '餐飲': '#ef4444', 
  '交通': '#f97316', 
  '購物': '#eab308', 
  '娛樂': '#22c55e', 
  '居家': '#3b82f6', 
  '進修': '#a855f7', 
  '家電': '#6366f1', 
  '薪資': '#10b981',
  '投資': '#06b6d4',
  '保險': '#f43f5e',
  '寵物': '#f59e0b'
};

// --- Helper Functions ---
const formatCurrency = (amount) => {
  return new Intl.NumberFormat('zh-TW', { style: 'currency', currency: 'TWD', minimumFractionDigits: 0 }).format(amount);
};

const getMonthString = (date) => {
  const d = new Date(date);
  const year = d.getFullYear();
  const month = String(d.getMonth() + 1).padStart(2, '0');
  return `${year}-${month}`;
};

const generateInstallmentPayments = (installment) => {
  const payments = [];
  let currentDate = new Date(installment.startDate);
  const monthlyAmount = Math.round(installment.totalAmount / installment.months);
  const remainder = installment.totalAmount - (monthlyAmount * installment.months);

  for (let i = 0; i < installment.months; i++) {
    payments.push({
      id: `${installment.id}-p${i+1}`,
      installmentId: installment.id,
      title: `${installment.title} (${i+1}/${installment.months})`,
      amount: i === 0 ? monthlyAmount + remainder : monthlyAmount,
      date: new Date(currentDate).toISOString().split('T')[0],
      monthStr: getMonthString(currentDate),
      type: 'expense',
      category: installment.category || '分期付款',
      isInstallmentPayment: true
    });
    currentDate.setMonth(currentDate.getMonth() + 1);
  }
  return payments;
};

// Generate a consistent vibrant color based on a string
const generateColorForString = (str) => {
  let hash = 0;
  for (let i = 0; i < str.length; i++) {
    hash = str.charCodeAt(i) + ((hash << 5) - hash);
  }
  const h = Math.abs(hash) % 360;
  return `hsl(${h}, 65%, 55%)`;
};

// --- Main Application Component ---
export default function App() {
  const [activeTab, setActiveTab] = useState('dashboard');
  const [currentMonthStr, setCurrentMonthStr] = useState(getMonthString(new Date()));
  
  // State
  const [user, setUser] = useState(null);
  const [isAuthLoading, setIsAuthLoading] = useState(true);
  const [transactions, setTransactions] = useState([]);
  const [installments, setInstallments] = useState([]);
  const [customTags, setCustomTags] = useState([]);
  
  // Editing State
  const [editingItem, setEditingItem] = useState(null); // { type: 'transaction' | 'installment', data: {...} }

  // --- Auth Effect ---
  useEffect(() => {
    if (!auth) {
      setIsAuthLoading(false);
      return;
    }

    const initAuth = async () => {
      try {
        if (initToken) {
          await signInWithCustomToken(auth, initToken);
        } else {
          await signInAnonymously(auth);
        }
      } catch (error) {
        console.error("Authentication failed:", error);
      }
    };
    initAuth();

    const unsubscribe = onAuthStateChanged(auth, (currentUser) => {
      setUser(currentUser);
      setIsAuthLoading(false);
    });

    return () => unsubscribe();
  }, []);

  // --- Data Fetching Effect ---
  useEffect(() => {
    if (!user || !db) return;

    const txRef = collection(db, 'artifacts', appId, 'users', user.uid, 'transactions');
    const unsubscribeTx = onSnapshot(txRef, 
      (snapshot) => setTransactions(snapshot.docs.map(doc => ({ id: doc.id, ...doc.data() }))),
      (error) => console.error("Error fetching transactions:", error)
    );

    const instRef = collection(db, 'artifacts', appId, 'users', user.uid, 'installments');
    const unsubscribeInst = onSnapshot(instRef, 
      (snapshot) => setInstallments(snapshot.docs.map(doc => ({ id: doc.id, ...doc.data() }))),
      (error) => console.error("Error fetching installments:", error)
    );

    const tagsRef = collection(db, 'artifacts', appId, 'users', user.uid, 'custom_tags');
    const unsubscribeTags = onSnapshot(tagsRef,
      (snapshot) => setCustomTags(snapshot.docs.map(doc => ({ id: doc.id, ...doc.data() }))),
      (error) => console.error("Error fetching custom tags:", error)
    );

    return () => {
      unsubscribeTx();
      unsubscribeInst();
      unsubscribeTags();
    };
  }, [user]);

  // Derived Data
  const allEffectiveTransactions = useMemo(() => {
    let combined = [...transactions];
    installments.forEach(inst => {
      combined = combined.concat(generateInstallmentPayments(inst));
    });
    return combined.sort((a, b) => new Date(b.date) - new Date(a.date));
  }, [transactions, installments]);

  const currentMonthData = useMemo(() => {
    const monthTx = allEffectiveTransactions.filter(t => getMonthString(t.date) === currentMonthStr);
    
    let income = 0;
    let expense = 0;
    
    monthTx.forEach(t => {
      if (t.type === 'income') income += t.amount;
      if (t.type === 'expense') expense += t.amount;
    });

    return { transactions: monthTx, income, expense, balance: income - expense };
  }, [allEffectiveTransactions, currentMonthStr]);

  // Tag Management
  const saveTagIfNew = async (tagText, tagType) => {
    if (!user || !db) return;
    if (!tagText.trim()) return;
    if (DEFAULT_CATEGORIES[tagType].includes(tagText)) return;
    const alreadyExists = customTags.some(t => t.text === tagText && t.type === tagType);
    if (alreadyExists) return;

    const id = Date.now().toString();
    try {
      await setDoc(doc(db, 'artifacts', appId, 'users', user.uid, 'custom_tags', id), { 
        id, text: tagText, type: tagType 
      });
    } catch (error) {
      console.error("Error saving custom tag:", error);
    }
  };

  const handleDeleteCustomTag = async (id) => {
    if (!user || !db) return;
    try {
      await deleteDoc(doc(db, 'artifacts', appId, 'users', user.uid, 'custom_tags', id));
    } catch (error) {
      console.error("Error deleting custom tag:", error);
    }
  };

  // Create Handlers
  const handleAddTransaction = async (newTx) => {
    if (!user || !db) return;
    const id = Date.now().toString();
    try {
      await setDoc(doc(db, 'artifacts', appId, 'users', user.uid, 'transactions', id), { ...newTx, id });
      await saveTagIfNew(newTx.category, newTx.type);
      setActiveTab('list');
    } catch (error) {
      console.error("Error adding transaction:", error);
    }
  };

  const handleAddInstallment = async (newInst) => {
    if (!user || !db) return;
    const id = Date.now().toString();
    try {
      await setDoc(doc(db, 'artifacts', appId, 'users', user.uid, 'installments', id), { ...newInst, id });
      await saveTagIfNew(newInst.category, 'installment');
      setActiveTab('installments');
    } catch (error) {
      console.error("Error adding installment:", error);
    }
  };

  // Update Handlers
  const handleUpdateTransaction = async (updatedTx) => {
    if (!user || !db) return;
    try {
      await updateDoc(doc(db, 'artifacts', appId, 'users', user.uid, 'transactions', updatedTx.id), updatedTx);
      await saveTagIfNew(updatedTx.category, updatedTx.type);
      setEditingItem(null);
      setActiveTab('list');
    } catch (error) {
      console.error("Error updating transaction:", error);
    }
  };

  const handleUpdateInstallment = async (updatedInst) => {
    if (!user || !db) return;
    try {
      await updateDoc(doc(db, 'artifacts', appId, 'users', user.uid, 'installments', updatedInst.id), updatedInst);
      await saveTagIfNew(updatedInst.category, 'installment');
      setEditingItem(null);
      setActiveTab('installments');
    } catch (error) {
      console.error("Error updating installment:", error);
    }
  };

  // Delete Handlers
  const handleDeleteTransaction = async (id) => {
    if (!user || !db) return;
    try {
      await deleteDoc(doc(db, 'artifacts', appId, 'users', user.uid, 'transactions', id));
    } catch (error) {
      console.error("Error deleting transaction:", error);
    }
  };

  const handleDeleteInstallment = async (id) => {
    if (!user || !db) return;
    try {
      await deleteDoc(doc(db, 'artifacts', appId, 'users', user.uid, 'installments', id));
    } catch (error) {
      console.error("Error deleting installment:", error);
    }
  };

  // Edit Trigger Handlers
  const handleEditRequest = (item, type = 'transaction') => {
    if (type === 'transaction') {
      if (item.isInstallmentPayment) {
        // Find the parent installment and edit that instead
        const parentInst = installments.find(i => i.id === item.installmentId);
        if (parentInst) {
          setEditingItem({ type: 'installment', data: parentInst });
        }
      } else {
        setEditingItem({ type: 'transaction', data: item });
      }
    } else {
      setEditingItem({ type: 'installment', data: item });
    }
    setActiveTab('edit');
  };

  if (isAuthLoading) {
    return (
      <div className="flex flex-col items-center justify-center h-screen bg-gray-50 text-indigo-600">
        <Loader2 className="w-10 h-10 animate-spin mb-4" />
        <p className="font-medium text-gray-600">連線至雲端資料庫...</p>
      </div>
    );
  }

  return (
    <div className="flex flex-col h-screen max-w-md mx-auto bg-gray-50 text-gray-800 font-sans shadow-xl overflow-hidden relative">
      <header className="bg-white px-6 py-4 shadow-sm z-10 flex justify-between items-center relative">
        <div className="flex items-center gap-2">
          <PieChartIcon className="w-6 h-6 text-indigo-600" />
          <h1 className="text-xl font-bold text-gray-800">記帳小幫手</h1>
        </div>
        
        <div className="flex items-center gap-3">
          {user ? (
            <div className="text-green-500 bg-green-50 p-1.5 rounded-full" title="已連線">
              <Cloud size={16} />
            </div>
          ) : (
            <div className="text-gray-400 bg-gray-100 p-1.5 rounded-full" title="離線模式">
              <CloudOff size={16} />
            </div>
          )}

          {(activeTab !== 'add' && activeTab !== 'edit') && (
            <input 
              type="month" 
              value={currentMonthStr}
              onChange={(e) => setCurrentMonthStr(e.target.value)}
              className="border-none bg-indigo-50 text-indigo-700 font-bold rounded-lg px-2 py-1 text-sm focus:ring-0 cursor-pointer w-32 text-center"
            />
          )}
        </div>
      </header>

      <main className="flex-1 overflow-y-auto p-4 pb-28">
        {!firebaseConfig && (
          <div className="bg-yellow-50 text-yellow-700 p-3 rounded-lg mb-4 text-xs font-medium text-center border border-yellow-200">
            提醒：目前處於無雲端環境，資料將無法保存。
          </div>
        )}
        
        {activeTab === 'dashboard' && <Dashboard data={currentMonthData} monthStr={currentMonthStr} />}
        {activeTab === 'calendar' && <CalendarView transactions={currentMonthData.transactions} currentMonthStr={currentMonthStr} onDelete={handleDeleteTransaction} onEdit={handleEditRequest} />}
        {activeTab === 'add' && <EntryForm mode="add" onAddTx={handleAddTransaction} onAddInst={handleAddInstallment} customTags={customTags} onDeleteCustomTag={handleDeleteCustomTag} />}
        {activeTab === 'edit' && editingItem && (
          <EntryForm 
            mode="edit" 
            initialData={editingItem.data}
            formType={editingItem.type}
            onUpdateTx={handleUpdateTransaction} 
            onUpdateInst={handleUpdateInstallment} 
            onCancel={() => {
              setEditingItem(null);
              setActiveTab(editingItem.type === 'installment' ? 'installments' : 'list');
            }}
            customTags={customTags} 
            onDeleteCustomTag={handleDeleteCustomTag} 
          />
        )}
        {activeTab === 'list' && <TransactionList transactions={currentMonthData.transactions} onDelete={handleDeleteTransaction} onEdit={handleEditRequest} />}
        {activeTab === 'installments' && <InstallmentManager installments={installments} currentMonthStr={currentMonthStr} onDelete={handleDeleteInstallment} onEdit={handleEditRequest} />}
      </main>

      <nav className="bg-white border-t border-gray-200 flex justify-around p-2 absolute bottom-0 w-full max-w-md pb-safe shadow-[0_-4px_6px_-1px_rgba(0,0,0,0.05)] z-20">
        <NavItem icon={<Home />} label="總覽" isActive={activeTab === 'dashboard'} onClick={() => { setEditingItem(null); setActiveTab('dashboard'); }} />
        <NavItem icon={<CalendarIcon />} label="日曆" isActive={activeTab === 'calendar'} onClick={() => { setEditingItem(null); setActiveTab('calendar'); }} />
        
        <div className="relative -top-6">
          <button 
            onClick={() => { setEditingItem(null); setActiveTab('add'); }}
            className={`bg-indigo-600 text-white p-4 rounded-full shadow-lg shadow-indigo-200 transform transition-transform ${activeTab === 'add' ? 'scale-110 bg-indigo-700' : 'hover:scale-105 active:scale-95'}`}
          >
            <PlusCircle className="w-7 h-7" />
          </button>
        </div>

        <NavItem icon={<List />} label="明細" isActive={activeTab === 'list'} onClick={() => { setEditingItem(null); setActiveTab('list'); }} />
        <NavItem icon={<CreditCard />} label="分期" isActive={activeTab === 'installments'} onClick={() => { setEditingItem(null); setActiveTab('installments'); }} />
      </nav>
    </div>
  );
}

// --- Sub-components ---

function Dashboard({ data, monthStr }) {
  const [year, month] = monthStr.split('-');

  return (
    <div className="space-y-6 animate-fade-in">
      <div className="bg-gradient-to-br from-indigo-500 to-purple-600 rounded-2xl p-6 text-white shadow-lg shadow-indigo-200">
        <p className="text-indigo-100 text-sm mb-1">{year}年 {month}月 結餘</p>
        <h2 className="text-4xl font-bold mb-6 tracking-tight">{formatCurrency(data.balance)}</h2>
        
        <div className="flex justify-between border-t border-indigo-400/30 pt-4">
          <div>
            <p className="text-indigo-100 text-xs flex items-center gap-1 mb-1">
              <ArrowUpCircle className="w-3 h-3" /> 總收入
            </p>
            <p className="font-semibold text-lg">{formatCurrency(data.income)}</p>
          </div>
          <div className="text-right">
            <p className="text-indigo-100 text-xs flex items-center gap-1 justify-end mb-1">
              <ArrowDownCircle className="w-3 h-3" /> 總支出
            </p>
            <p className="font-semibold text-lg">{formatCurrency(data.expense)}</p>
          </div>
        </div>
      </div>

      <ExpensePieChart transactions={data.transactions} totalExpense={data.expense} />
    </div>
  );
}

function ExpensePieChart({ transactions, totalExpense }) {
  const [isExpanded, setIsExpanded] = useState(false);

  const expenseByCategory = useMemo(() => {
    const categories = {};
    transactions.forEach(tx => {
      if (tx.type === 'expense') {
        categories[tx.category] = (categories[tx.category] || 0) + tx.amount;
      }
    });
    
    return Object.keys(categories)
      .map(name => ({
        name,
        amount: categories[name],
        percent: (categories[name] / totalExpense) * 100,
        color: CATEGORY_COLORS[name] || generateColorForString(name)
      }))
      .sort((a, b) => b.amount - a.amount);
  }, [transactions, totalExpense]);

  if (totalExpense === 0) {
    return (
      <div className="bg-white rounded-xl p-6 shadow-sm border border-gray-100 text-center">
        <h3 className="text-gray-600 font-bold mb-2 text-sm text-left">支出結構明細</h3>
        <div className="w-32 h-32 mx-auto rounded-full border-8 border-gray-100 flex items-center justify-center my-4">
          <p className="text-gray-400 text-xs">無支出紀錄</p>
        </div>
      </div>
    );
  }

  const radius = 15.91549430918954;
  let cumulativePercent = 0;
  const displayCategories = isExpanded ? expenseByCategory : expenseByCategory.slice(0, 4);

  return (
    <div className="bg-white rounded-2xl p-6 shadow-sm border border-gray-100">
      <h3 className="text-gray-800 font-bold mb-6 text-base">本月支出明細排行</h3>
      
      <div className="flex justify-center mb-8">
        <div className="relative w-40 h-40">
          <svg viewBox="0 0 42 42" className="w-full h-full transform -rotate-90 drop-shadow-sm">
            <circle cx="21" cy="21" r={radius} fill="transparent" stroke="#f1f5f9" strokeWidth="6" />
            {expenseByCategory.map((category) => {
              const dasharray = `${category.percent} ${100 - category.percent}`;
              const dashoffset = 100 - cumulativePercent;
              cumulativePercent += category.percent;
              
              return (
                <circle 
                  key={category.name}
                  cx="21" cy="21" r={radius} 
                  fill="transparent" 
                  stroke={category.color} 
                  strokeWidth="6" 
                  strokeDasharray={dasharray} 
                  strokeDashoffset={dashoffset}
                  className="transition-all duration-500 ease-out"
                />
              );
            })}
          </svg>
          <div className="absolute inset-0 flex flex-col items-center justify-center pointer-events-none">
            <span className="text-xs font-medium text-gray-500 mb-0.5">總支出</span>
            <span className="text-sm font-bold text-gray-800">
              ${totalExpense > 9999 ? `${(totalExpense / 1000).toFixed(1)}k` : totalExpense}
            </span>
          </div>
        </div>
      </div>

      <div className="space-y-3">
        {displayCategories.map((category) => (
          <div key={category.name} className="flex items-center justify-between p-2 rounded-lg hover:bg-gray-50 transition-colors">
            <div className="flex items-center gap-3">
              <span className="w-3.5 h-3.5 rounded-full shadow-sm" style={{ backgroundColor: category.color }}></span>
              <span className="text-gray-700 font-bold text-sm">{category.name}</span>
            </div>
            <div className="flex items-center gap-4">
              <span className="text-gray-800 font-bold tracking-tight">
                {formatCurrency(category.amount)}
              </span>
              <span className="text-gray-400 font-medium text-xs w-8 text-right bg-gray-100 rounded-md py-0.5">
                {Math.round(category.percent)}%
              </span>
            </div>
          </div>
        ))}

        {expenseByCategory.length > 4 && (
          <button
            onClick={() => setIsExpanded(!isExpanded)}
            className="w-full mt-2 pt-3 border-t border-dashed border-gray-200 text-indigo-500 font-bold text-sm flex items-center justify-center gap-1 hover:text-indigo-600 transition-colors"
          >
            {isExpanded ? (
              <>收起明細清單 <ChevronUp size={16} /></>
            ) : (
              <>查看其餘 {expenseByCategory.length - 4} 項類別明細 <ChevronDown size={16} /></>
            )}
          </button>
        )}
      </div>
    </div>
  );
}

function CalendarView({ transactions, currentMonthStr, onDelete, onEdit }) {
  const [selectedDate, setSelectedDate] = useState(new Date().toISOString().split('T')[0]);

  const [yearStr, monthStr] = currentMonthStr.split('-');
  const year = parseInt(yearStr, 10);
  const month = parseInt(monthStr, 10) - 1;

  const calendarDays = useMemo(() => {
    const firstDay = new Date(year, month, 1);
    const lastDay = new Date(year, month + 1, 0);
    const startingDayOfWeek = firstDay.getDay();
    const totalDays = lastDay.getDate();

    const days = [];
    
    for (let i = 0; i < startingDayOfWeek; i++) {
      days.push(null);
    }

    for (let i = 1; i <= totalDays; i++) {
      const dateStr = `${year}-${String(month + 1).padStart(2, '0')}-${String(i).padStart(2, '0')}`;
      
      const dayTxs = transactions.filter(t => t.date === dateStr);
      let income = 0;
      let expense = 0;
      dayTxs.forEach(t => {
        if (t.type === 'income') income += t.amount;
        if (t.type === 'expense') expense += t.amount;
      });

      days.push({
        day: i,
        date: dateStr,
        transactions: dayTxs,
        expense,
        income,
        net: income - expense
      });
    }

    return days;
  }, [transactions, year, month]);

  const selectedTransactions = useMemo(() => {
    return transactions.filter(t => t.date === selectedDate);
  }, [transactions, selectedDate]);

  const todayStr = new Date().toISOString().split('T')[0];

  return (
    <div className="space-y-6 animate-fade-in pb-4">
      <div className="bg-white rounded-2xl shadow-sm border border-gray-100 p-4">
        <div className="grid grid-cols-7 gap-1 text-center mb-2">
          {['日', '一', '二', '三', '四', '五', '六'].map(d => (
            <div key={d} className="text-xs font-bold text-gray-400 py-1">{d}</div>
          ))}
        </div>

        <div className="grid grid-cols-7 gap-1 gap-y-2">
          {calendarDays.map((dayData, index) => {
            if (!dayData) {
              return <div key={`empty-${index}`} className="h-16 rounded-xl bg-gray-50/50"></div>;
            }

            const isSelected = dayData.date === selectedDate;
            const isToday = dayData.date === todayStr;
            const hasExpense = dayData.expense > 0;
            const hasIncome = dayData.income > 0;

            return (
              <button
                key={dayData.date}
                onClick={() => setSelectedDate(dayData.date)}
                className={`
                  h-16 rounded-xl flex flex-col items-center pt-1 pb-1 px-0.5 relative transition-all
                  ${isSelected ? 'bg-indigo-50 border-2 border-indigo-500 shadow-sm' : 'bg-white border-2 border-transparent hover:bg-gray-50'}
                `}
              >
                <span className={`text-sm font-bold ${isToday ? 'bg-indigo-600 text-white w-6 h-6 rounded-full flex items-center justify-center' : 'text-gray-700'}`}>
                  {dayData.day}
                </span>
                
                <div className="mt-auto flex flex-col items-center w-full">
                  {hasIncome && <span className="text-[9px] font-bold text-green-500 leading-none mb-0.5">+{dayData.income > 9999 ? '9k+' : dayData.income}</span>}
                  {hasExpense && <span className="text-[9px] font-bold text-red-500 leading-none">-{dayData.expense > 9999 ? '9k+' : dayData.expense}</span>}
                </div>
              </button>
            );
          })}
        </div>
      </div>

      <div>
        <h3 className="text-sm font-bold text-gray-600 mb-3 px-1 flex items-center gap-2">
          <CalendarIcon size={16} />
          {selectedDate.replace(/-/g, '/')} 明細
        </h3>
        
        {selectedTransactions.length === 0 ? (
          <div className="bg-white rounded-2xl border border-gray-100 border-dashed p-8 text-center text-gray-400">

              <div 
                key={tx.id} 
                onClick={() => onEdit(tx, 'transaction')}

                <div className="flex items-center gap-3">
                  <div className="flex-shrink-0">
                    <div 
                      className="w-10 h-10 rounded-full flex items-center justify-center text-white font-bold shadow-sm"
                      style={{ backgroundColor: tx.isInstallmentPayment ? '#a855f7' : tx.type === 'income' ? '#22c55e' : (CATEGORY_COLORS[tx.category] || generateColorForString(tx.category)) }}

                      {tx.isInstallmentPayment ? <CreditCard size={18} /> : tx.type === 'income' ? <ArrowUpCircle size={18} /> : <Tag size={18} />}
                    </div>
                  </div>
                  <div className="flex flex-col">
                    <p className="font-bold text-gray-800 text-sm leading-tight flex items-center gap-1">
                      {tx.title} 
                    </p>
                    <div className="flex items-center gap-1 mt-1">
                      <span className="text-[10px] bg-gray-100 text-gray-500 px-1.5 py-0.5 rounded-md">
                        {tx.category}
                      </span>
                      {tx.isInstallmentPayment && <span className="text-[10px] text-purple-500 font-medium">自動分攤</span>}
                    </div>
                  </div>
                </div>
                <div className="flex items-center gap-2">
                  <span className={`font-bold tracking-tight ${tx.type === 'income' ? 'text-green-600' : 'text-gray-800'}`}>
                    {tx.type === 'income' ? '+' : '-'}{formatCurrency(tx.amount)}
                  </span>
                  
                  <div className="flex items-center">
                     {!tx.isInstallmentPayment ? (
                       <button onClick={(e) => { e.stopPropagation(); onDelete(tx.id); }} className="text-gray-300 hover:text-red-500 transition-colors p-1.5 ml-1 bg-white rounded-full shadow-sm border border-gray-100">
                         <Trash2 size={14} />
                       </button>
                     ) : (
                       <div className="p-1.5 ml-1 text-purple-300">
                         <Pencil size={14} />
                       </div>
                     )}
                  </div>
                </div>
              </div>
            ))}
          </div>
        )}
      </div>
    </div>
  );
}

// 支援「新增」與「編輯」雙模式的表單元件
function EntryForm({ mode = 'add', initialData = null, formType = 'transaction', onAddTx, onAddInst, onUpdateTx, onUpdateInst, onCancel, customTags, onDeleteCustomTag }) {
  const [type, setType] = useState('expense');
  const [amount, setAmount] = useState('');
  const [title, setTitle] = useState('');
  const [category, setCategory] = useState(DEFAULT_CATEGORIES['expense'][0]);
  const [date, setDate] = useState(new Date().toISOString().split('T')[0]);
  const [months, setMonths] = useState(3);

  // 初始化資料 (如果是編輯模式)
  useEffect(() => {
    if (mode === 'edit' && initialData) {
      if (formType === 'installment') {
        setType('installment');
        setAmount(initialData.totalAmount.toString());
        setMonths(initialData.months.toString());
        setTitle(initialData.title);
        setCategory(initialData.category);
        setDate(initialData.startDate);
      } else {
        setType(initialData.type);
        setAmount(initialData.amount.toString());
        setTitle(initialData.title);
        setCategory(initialData.category);
        setDate(initialData.date);
      }
    }
  }, [mode, initialData, formType]);

  const handleSubmit = (e) => {
    e.preventDefault();
    if (!amount || !title || !category) return;

    if (mode === 'add') {
      if (type === 'installment') {
        onAddInst({ title, totalAmount: Number(amount), months: Number(months), startDate: date, category: category.trim() });
      } else {
        onAddTx({ type, amount: Number(amount), title, category: category.trim(), date });
      }
    } else if (mode === 'edit') {
      if (type === 'installment') {
        onUpdateInst({ ...initialData, title, totalAmount: Number(amount), months: Number(months), startDate: date, category: category.trim() });
      } else {
        onUpdateTx({ ...initialData, type, amount: Number(amount), title, category: category.trim(), date });
      }
    }
  };

  const handleTypeChange = (newType) => {
    setType(newType);
    setCategory(DEFAULT_CATEGORIES[newType][0]);
  };

  const currentCustomTags = customTags.filter(t => t.type === type);

  return (
    <div className="bg-white rounded-2xl shadow-sm border border-gray-100 p-6 animate-fade-in relative">
      <div className="flex justify-between items-center mb-6">
        <h2 className="text-xl font-bold text-gray-800 w-full text-center">
          {mode === 'edit' ? (type === 'installment' ? '編輯分期專案' : '編輯紀錄') : '新增紀錄'}
        </h2>
      </div>
      
      {mode === 'add' && (
        <div className="flex bg-gray-100 rounded-xl p-1 mb-6">
          <button onClick={() => handleTypeChange('expense')} type="button" className={`flex-1 py-2 text-sm font-bold rounded-lg transition-all ${type === 'expense' ? 'bg-white text-red-500 shadow-sm' : 'text-gray-500 hover:text-gray-700'}`}>一般支出</button>
          <button onClick={() => handleTypeChange('income')} type="button" className={`flex-1 py-2 text-sm font-bold rounded-lg transition-all ${type === 'income' ? 'bg-white text-green-600 shadow-sm' : 'text-gray-500 hover:text-gray-700'}`}>收入</button>
          <button onClick={() => handleTypeChange('installment')} type="button" className={`flex-1 py-2 text-sm font-bold rounded-lg transition-all ${type === 'installment' ? 'bg-white text-purple-600 shadow-sm' : 'text-gray-500 hover:text-gray-700'}`}>信用卡分期</button>
        </div>
      )}

      {mode === 'edit' && type === 'installment' && (
        <div className="bg-purple-50 text-purple-700 text-xs font-bold px-3 py-2 rounded-lg mb-6 text-center">
          💡 修改母專案將自動重新計算所有月份的分攤金額
        </div>
      )}

      <form onSubmit={handleSubmit} className="space-y-4">
        <div>
          <label className="block text-xs font-semibold text-gray-600 mb-1">
            {type === 'installment' ? '總金額 (將自動平均分攤)' : '金額'}
          </label>
          <div className="relative">
            <span className="absolute left-4 top-3.5 text-gray-400 font-bold">$</span>
            <input 
              type="number" 
              value={amount}
              onChange={(e) => setAmount(e.target.value)}
              className="w-full pl-8 pr-4 py-3 bg-gray-50 border border-gray-200 rounded-xl focus:ring-2 focus:ring-indigo-500 focus:border-indigo-500 transition-all font-semibold text-gray-800 text-lg outline-none"
              placeholder="0" required min="1"
            />
          </div>
        </div>

        {type === 'installment' && (
          <div>
            <label className="block text-xs font-semibold text-gray-600 mb-1">分期期數 (月)</label>
            <select 
              value={months}
              onChange={(e) => setMonths(e.target.value)}
              className="w-full px-4 py-3 bg-gray-50 border border-gray-200 rounded-xl focus:ring-2 focus:ring-indigo-500 focus:border-indigo-500 transition-all font-medium text-gray-700 outline-none"
            >
              {[3, 6, 12, 18, 24, 30].map(m => <option key={m} value={m}>{m} 期</option>)}
            </select>
          </div>
        )}

        <div>
          <label className="block text-xs font-semibold text-gray-600 mb-1">項目名稱</label>
          <div className="relative">
            <FileText className="absolute left-3 top-3.5 text-gray-400 w-5 h-5" />
            <input 
              type="text" 
              value={title}
              onChange={(e) => setTitle(e.target.value)}
              className="w-full pl-10 pr-4 py-3 bg-gray-50 border border-gray-200 rounded-xl focus:ring-2 focus:ring-indigo-500 focus:border-indigo-500 transition-all font-medium text-gray-800 outline-none"
              placeholder={type === 'installment' ? '例如：iPhone 15' : '例如：午餐便當'} required
            />
          </div>
        </div>

        <div>
          <label className="block text-xs font-semibold text-gray-600 mb-1">分類標籤 (可點選或自訂輸入)</label>
          <div className="relative">
            <Tag className="absolute left-3 top-3.5 text-gray-400 w-5 h-5" />
            <input 
              type="text" 
              value={category}
              onChange={(e) => setCategory(e.target.value)}
              className="w-full pl-10 pr-4 py-3 bg-gray-50 border border-gray-200 rounded-xl focus:ring-2 focus:ring-indigo-500 focus:border-indigo-500 transition-all font-medium text-gray-800 outline-none"
              placeholder="點選下方標籤或自行輸入新標籤" required
            />
          </div>
          
          <div className="flex flex-wrap gap-2 mt-3">
            {DEFAULT_CATEGORIES[type].map(c => (
              <button
                key={c} type="button" onClick={() => setCategory(c)}
                className={`px-3 py-1.5 text-xs font-medium rounded-full border transition-colors ${category === c ? 'bg-gray-800 border-gray-800 text-white' : 'bg-white border-gray-200 text-gray-600 hover:bg-gray-50'}`}
              >
                {c}
              </button>
            ))}

            {currentCustomTags.map(tag => (
              <div 
                key={tag.id} 
                className={`flex items-center border rounded-full overflow-hidden transition-colors ${category === tag.text ? 'bg-indigo-600 border-indigo-600 text-white' : 'bg-indigo-50 border-indigo-100 text-indigo-700'}`}
              >
                <button type="button" onClick={() => setCategory(tag.text)} className="px-3 py-1.5 text-xs font-bold hover:bg-black/5 transition-colors">
                  {tag.text}
                </button>
                <button
                  type="button"
                  onClick={(e) => { e.stopPropagation(); onDeleteCustomTag(tag.id); }}
                  className={`px-1.5 py-1.5 transition-colors ${category === tag.text ? 'hover:bg-indigo-700 text-indigo-200 hover:text-white' : 'hover:bg-indigo-200 text-indigo-400 hover:text-indigo-800'}`}
                >
                  <X size={14} strokeWidth={3} />
                </button>
              </div>
            ))}
          </div>
        </div>

        <div>
           <label className="block text-xs font-semibold text-gray-600 mb-1 mt-4">
             {type === 'installment' ? '首期日期' : '日期'}
           </label>
           <div className="relative">
             <CalendarIcon className="absolute left-3 top-3.5 text-gray-400 w-5 h-5" />
             <input 
               type="date" 
               value={date}
               onChange={(e) => setDate(e.target.value)}
               className="w-full pl-10 pr-3 py-3 bg-gray-50 border border-gray-200 rounded-xl focus:ring-2 focus:ring-indigo-500 focus:border-indigo-500 transition-all text-sm font-medium text-gray-700 outline-none" required
             />
           </div>
        </div>

        <div className="flex gap-3 mt-8 pt-4 border-t border-gray-100">
          {mode === 'edit' && (
            <button 
              type="button"
              onClick={onCancel}
              className="flex-1 bg-gray-100 hover:bg-gray-200 text-gray-700 font-bold py-4 rounded-xl transition-colors"
            >
              取消
            </button>
          )}
          <button 
            type="submit" 
            className={`flex-[2] text-white font-bold py-4 rounded-xl shadow-md transition-colors flex justify-center items-center gap-2 ${mode === 'edit' ? 'bg-green-600 hover:bg-green-700' : 'bg-indigo-600 hover:bg-indigo-700'}`}
          >
            {mode === 'edit' ? <Pencil size={20} /> : <PlusCircle size={20} />}
            {mode === 'edit' ? '更新紀錄' : '儲存紀錄至雲端'}
          </button>
        </div>
      </form>
    </div>
  );
}

function TransactionList({ transactions, onDelete, onEdit }) {

        <List className="w-16 h-16 mb-4 opacity-20 text-indigo-300" />

        <p className="text-xs mt-2">點擊下方「+」按鈕開始記帳</p>
      </div>
    );
  }

  const grouped = transactions.reduce((acc, tx) => {
    if (!acc[tx.date]) acc[tx.date] = [];
    acc[tx.date].push(tx);
    return acc;
  }, {});

  return (
    <div className="space-y-6 animate-fade-in pb-4">
      {sortedDates.map(date => {
        const dailyTotal = grouped[date].reduce((sum, tx) => {
          return tx.type === 'income' ? sum + tx.amount : sum - tx.amount;

          <div key={date}>
            <div className="flex justify-between items-end mb-2 px-1">

            <div className="bg-white rounded-2xl shadow-sm border border-gray-100 overflow-hidden">

                <div 
                  key={tx.id} 
                  onClick={() => onEdit(tx, 'transaction')}
                  className={`p-4 flex items-center justify-between cursor-pointer hover:bg-indigo-50 transition-colors ${idx !== grouped[date].length - 1 ? 'border-b border-gray-50' : ''}`}

function InstallmentManager({ installments, currentMonthStr, onDelete, onEdit }) {
  if (installments.length === 0) {
    return (
      <div className="flex flex-col items-center justify-center py-24 text-gray-400 animate-fade-in">
        <CreditCard className="w-16 h-16 mb-4 opacity-20 text-purple-300" />
        <p className="font-medium text-gray-500">目前沒有進行中的分期付款。</p>
        <p className="text-xs mt-2 text-center px-6">在新增頁面選擇「信用卡分期」，系統會自動為您計算每個月的應繳金額。</p>
      </div>
    );
  }

  return (
    <div className="space-y-4 animate-fade-in pb-4">
      <div className="flex items-center gap-2 mb-4">
        <h2 className="text-gray-800 font-bold text-lg">進行中分期專案</h2>
        <span className="bg-purple-100 text-purple-700 text-xs font-bold px-2 py-0.5 rounded-full">
          共 {installments.length} 筆
        </span>
      </div>

      {installments.map(inst => {
        const payments = generateInstallmentPayments(inst);
        let paidCount = 0;
        let paidAmount = 0;

        payments.forEach((p, idx) => {
          if (p.monthStr <= currentMonthStr) {
            paidCount = idx + 1;
            paidAmount += p.amount;
          }
        });

        if (currentMonthStr < payments[0].monthStr) {
          paidCount = 0;
          paidAmount = 0;
        }
        if (currentMonthStr > payments[payments.length - 1].monthStr) {
          paidCount = inst.months;
          paidAmount = inst.totalAmount;
        }

        const remainingAmount = inst.totalAmount - paidAmount;
        const isFinished = paidCount >= inst.months;
        const progressPercentage = (paidCount / inst.months) * 100;

        return (
          <div 
            key={inst.id} 
            onClick={() => onEdit(inst, 'installment')}
            className={`bg-white rounded-2xl shadow-sm border p-5 relative overflow-hidden transition-all cursor-pointer hover:shadow-md ${isFinished ? 'border-green-200 bg-green-50/30' : 'border-purple-100 hover:border-purple-300'}`}
          >
            <div className="flex justify-between items-start mb-3">
              <div className="flex-1">
                <h3 className="font-bold text-gray-800 text-lg leading-tight flex items-center gap-2">
                  {inst.title}
                  <Pencil size={14} className="text-gray-300" />
                </h3>
                <div className="text-xs font-medium text-gray-500 mt-2 flex flex-wrap items-center gap-2">
                  <span className="inline-block px-2 py-0.5 bg-gray-100 text-gray-600 rounded-md">
                    {inst.category}
                  </span>
                  <span>總額 {formatCurrency(inst.totalAmount)}</span>
                  
                  {!isFinished && remainingAmount > 0 && (
                    <span className="text-red-500 font-bold ml-auto bg-red-50 px-2 py-0.5 rounded-md">
                      待繳 {formatCurrency(remainingAmount)}
                    </span>
                  )}
                </div>
              </div>
              <button 
                onClick={(e) => { e.stopPropagation(); onDelete(inst.id); }} 
                className="text-gray-300 hover:text-red-500 bg-gray-50 p-2 rounded-full transition-colors ml-3 border border-gray-100"
              >
                <Trash2 size={16} />
              </button>
            </div>

            <div className="mt-5">
              <div className="flex justify-between text-xs font-bold mb-2">
                <span className={isFinished ? 'text-green-600' : 'text-purple-600'}>
                  {isFinished ? '🎉 恭喜！已全數結清' : `繳款進度：${paidCount} / ${inst.months} 期`}
                </span>
                <span className="text-gray-600">
                  每期 {formatCurrency(Math.round(inst.totalAmount / inst.months))}
                </span>
              </div>
              <div className="w-full bg-gray-100 rounded-full h-3 overflow-hidden">
                <div 
                  className={`h-full transition-all duration-1000 ease-out relative ${isFinished ? 'bg-green-500' : 'bg-gradient-to-r from-purple-400 to-indigo-500'}`} 
                  style={{ width: `${progressPercentage}%` }}
                >
                  <div className="absolute top-0 right-0 bottom-0 left-0 bg-white opacity-20" style={{ background: 'linear-gradient(90deg, transparent, rgba(255,255,255,0.4), transparent)', transform: 'translateX(-100%)', animation: 'shimmer 2s infinite' }}></div>
                </div>
              </div>
            </div>
          </div>
        );
      })}
    </div>
  );
}
